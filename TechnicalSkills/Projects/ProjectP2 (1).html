How does the flight duration in minutes relate to the number of employees scheduled on a flight?¶

In [83]:
q_sch = '''SELECT 
COUNT(es.EmployeeID) AS EmployeesScheduled,
CASE 
WHEN f.FlightArrival >= f.FlightDeparture THEN 
DATEDIFF(MINUTE, CAST('1900-01-01' AS datetime) + CAST(f.FlightDeparture AS datetime), CAST('1900-01-01' AS datetime) + CAST(f.FlightArrival AS datetime))
ELSE 
DATEDIFF(MINUTE, CAST('1900-01-01' AS datetime) + CAST(f.FlightDeparture AS datetime), DATEADD(DAY, 1, CAST('1900-01-01' AS datetime) + CAST(f.FlightArrival AS datetime)))
END AS FlightDurationMins
FROM 
Flight f
INNER JOIN FlightOrigin fo ON f.FlightID = fo.FlightID
INNER JOIN FlightDestination fd ON f.FlightID = fd.FlightID
INNER JOIN EmployeeSchedule es ON f.FlightID = es.FlightID
GROUP BY f.FlightArrival,
f.FlightDeparture
;'''

q_ticket = '''SELECT 
t.FlightID,
COUNT(t.FlightID) AS TotalTicketQty,
COUNT(es.EmployeeID) AS EmployeesScheduled,
SUM(t.TicketPrice) AS TotalTicketRevenue,
f.FlightAircraft,
CASE 
WHEN f.FlightArrival >= f.FlightDeparture THEN 
DATEDIFF(MINUTE, CAST('1900-01-01' AS datetime) + CAST(f.FlightDeparture AS datetime), CAST('1900-01-01' AS datetime) + CAST(f.FlightArrival AS datetime))
ELSE 
DATEDIFF(MINUTE, CAST('1900-01-01' AS datetime) + CAST(f.FlightDeparture AS datetime), DATEADD(DAY, 1, CAST('1900-01-01' AS datetime) + CAST(f.FlightArrival AS datetime)))
END AS FlightDurationMins,
t.TicketDate,
fo.AirportID AS 'AirportOrigin',
fd.AirportID AS 'AirportDestination'
FROM 
Ticket t
INNER JOIN Flight f ON t.FlightID = f.FlightID
INNER JOIN FlightOrigin fo ON f.FlightID = fo.FlightID
INNER JOIN FlightDestination fd ON f.FlightID = fd.FlightID
INNER JOIN EmployeeSchedule es ON f.FlightID = es.FlightID
GROUP BY 
t.FlightID,
f.FlightAircraft,
t.TicketDate,
fo.AirportID,
fd.AirportID,
f.FlightDeparture,
f.FlightArrival
ORDER BY 
t.FlightID ASC
;'''
df_ticket = pd.read_sql(q_ticket,connection)
